theorem lcm_mono {s₁ s₂ : Multiset α} (h : s₁ ⊆ s₂) : s₁.lcm ∣ s₂.lcm := by
  refine Multiset.induction_on s₁ ?base ?step

    have hs : s ⊆ t := by
