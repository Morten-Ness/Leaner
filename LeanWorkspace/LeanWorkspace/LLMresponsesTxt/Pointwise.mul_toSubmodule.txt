    (S T : Subalgebra R A) : (Subalgebra.toSubmodule S) * (Subalgebra.toSubmodule T)
        = Subalgebra.toSubmodule (S ⊔ T) := by
  rw [show (Subalgebra.toSubmodule S) * (Subalgebra.toSubmodule T) =
      Subalgebra.toSubmodule (S * T) by rfl]
  congr
