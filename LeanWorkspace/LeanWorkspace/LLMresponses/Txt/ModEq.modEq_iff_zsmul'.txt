  constructor

    rcases h with ⟨m, n, hmn⟩

        b = -(n • p) + (n • p + b) := by abel
        _ = -(n • p) + (m • p + a) := by rw [hmn]

      b - a = ((m - n) • p + a) - a := by rw [h1]
      _ = (m - n) • p := by abel
