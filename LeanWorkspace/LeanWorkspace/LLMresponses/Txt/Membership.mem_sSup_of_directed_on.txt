theorem mem_sSup_of_directed_on {S : Set (Subsemigroup M)} (hS : DirectedOn (· ≤ ·) S) {x : M} :
    x ∈ sSup S ↔ ∃ s ∈ S, x ∈ s := by
  classical
  simpa [Subsemigroup.sSup_eq_closure, Subsemigroup.mem_closure] using
    (show (∃ y : List M, (∀ z, z ∈ y → ∃ s ∈ S, z ∈ s) ∧ y.prod = x) ↔ ∃ s ∈ S, x ∈ s by
      constructor
      · rintro ⟨y, hy, rfl⟩
        induction y with
        | nil =>
            refine ⟨⊥, ?_, by simp⟩
            rcases Set.nonempty_of_mem hS with ⟨s, hs⟩
            exact by simpa using (show (⊥ : Subsemigroup M) ≤ s from bot_le)
        | cons a y ih =>
            rcases ih ?_ with ⟨s, hsS, hsy⟩
            · rcases hy a (by simp) with ⟨sa, hsaS, hasa⟩
              rcases hS hsS hsaS with ⟨t, htS, hst, hsat⟩

            · intro z hz
              exact hy z (by simp [hz])

        intro z hz
        simp only [List.mem_singleton] at hz
        subst hz
        exact ⟨s, hs, hx⟩)
