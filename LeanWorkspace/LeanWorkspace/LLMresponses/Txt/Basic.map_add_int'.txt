    (f : F) (x : G) (n : ℤ) : f (x + n) = f x + n • b := by
  cases n using Int.inductionOn with
  | hz =>
      simp

      simpa [Int.ofNat_eq_coe, add_assoc, add_left_comm, add_comm, zsmul_intNatEq_natCast_mul] using

          rw [show x + (n : ℤ) + 1 = (x + (n : ℤ)) + 1 by abel]
